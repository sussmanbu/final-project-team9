server <- function(input, output) {
    # Debug: observe the type and value of input$yearSelect
    observe({
        print(paste("Selected year:", input$yearSelect))
        print(paste("Type of input$yearSelect:", class(input$yearSelect)))
    })

    output$map <- renderLeaflet({
        # Make sure input$yearSelect is not NULL and data type is consistent
        if (is.null(input$yearSelect)) {
            stop("Year selection is NULL")
        }

        # Convert input to numeric if it's not, assuming 'year' in 'rent_data' is numeric
        year_input <- as.numeric(input$yearSelect)

        # Filter data
        data_filtered <- rent_data %>%
            filter(year == year_input)

        # Check if data filtering works
        print(paste("Number of rows after filter:", nrow(data_filtered)))

        # Proceed with creating the map
        leaflet(data = data_filtered) %>%
            addTiles() %>%
            addCircleMarkers(
                ~longitude, ~latitude, 
                radius = ~sqrt(median_rent),
                popup = ~paste("Median Rent: $", median_rent),
                fillColor = "#ff7800", color = "#FFFFFF",
                fillOpacity = 0.7, weight = 1
            )
    })
}
